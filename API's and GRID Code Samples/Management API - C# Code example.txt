The following is C# code that can create a user with the BSIDCA API:


            BSIDCA ca = new BSIDCA();
            ca.Url = "http://localhost/BSIDCA/BSIDCA.asmx";
            String challenge;
            if (ca.Connect("operator@credentials.com", "password_or_otp", null, out challenge) == ReturnCode.AUTH_SUCCESS)
            {
                String organization = "account_name";
                
                User u = new User();
                u.UserName = "BSIDCA User";
                ca.AddUser(u, organization);
            }
